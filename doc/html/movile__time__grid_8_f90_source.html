106  subroutine utm_2_ll(utmx,utmy,utmz,lat,long)
107  implicit none
108  integer,intent (IN):: utmZ
109  real,intent (IN):: utmx,utmy
110  real,intent (OUT) ::lat,long
111  real:: utmym,dlong,dlongp,dlongx
112  real:: DEGRAD,latx
113 !
114 ! THIS SUBROUTINE CONVERTS from UTM coordinates
115 !
116  DATA degrad/0.017453/
117 
118  latx = utmy / 111.
119  dlongx = (utmx - 500.) / ((111.226 + 0.0053 * latx) * &
120  (cos(degrad * latx)))
121  utmym = utmy - (3187. * sin(2. * degrad * latx) * &
122  (1. -cos(degrad * dlongx)))
123  lat = (utmym - 2.41 - 0.00903 * latx * latx) / 110.270
124  dlongp = (utmx - 500.) / ((111.226 + 0.0053 * lat) * &
125  (cos(degrad * lat)))
126  long = -(180 - (6 * utmz) + 3 - dlongp )
127 
128  end subroutine utm_2_ll

302 subroutine genera_malla
303  implicit none
304  integer:: l,i,n,m
305  integer:: indx
306  real :: sl,fcorr,ffr
307  real :: temp
308  do n = ntd ,1,-1 !Main LOOP initialization
309  do m = 1,nint
310  if(isource(n).eq.idsrc(m).and.isrc(m).lt.2 .and.igeo(m).lt.2) then
311  indx = idcell(m)
312  do l=1,nhr
313  ! for EPA
314  efact(1)= emisfac2(itype(n),vel(l,n),vele ,ehc )
315  efact(2)= emisfac2(itype(n),vel(l,n),vele ,eco )
316  efact(3)= emisfac2(itype(n),vel(l,n),vele ,eno )
317  ! Emissions factor for cold start
318  efac2(1)= emisfac2(itype(n),vel(l,n),vele2,ehc2)
319  efac2(2)= emisfac2(itype(n),vel(l,n),vele2,eco2)
320  efac2(3)= emisfac2(itype(n),vel(l,n),vele2,eno2)
321  if (itype(n).ge. 14 ) then
322  efact(4) =efact(1)
323  efac2(4) =efac2(1)
324  efact(1) = 0.0
325  efac2(1) = 0.0
326  else
327  efact(4) = 0.0
328  efac2(4) = 0.0
329  end if
330 !..
331 ! ---------- Localization of the viality lenght ----------
332 !..
333  call viality(igeo(m),idsrc(m),igeo2,idsrc2,kstype,slen &
334  ,fcor,start,l,fcorr,ffr,sl)
335  sl = sl*cw(m)
336 ! ---------- Computation of the emissions for each specie
337  do i = 1,nspc ! 1 HC(gasoline), 2 CO, 3 NOx, 4 HC(Diesel),5 SO2
338 !..
339  temp =(autn(l,n)*sl*efact(i)*(1.0-ffr)+ &
340  autn(l,n)*sl*efac2(i)*ffr)*fcorr
341 !.. Xing is 7.5% of the emission of segment viality
342  if (isrc(m).eq.2) temp=temp*0.075
343 !..
344  emision(indx,l,i,id_time_period(n))= temp/3600.0 &
345  + emision(indx,l,i,id_time_period(n))
346 !..
347 ! ---------- Computation of dayly emissions EDAY ----------
348 !
349  eday(indx,i,id_time_period(n),itype(n)-10,2)= temp/1000 &
350  + eday(indx,i,id_time_period(n),itype(n)-10,2)
351  end do ! i nspc
352  end do ! m nint
353  end if
354  end do ! ntd
355  end do
356 end subroutine genera_malla

617  real function emisfac2(ncartype,velocity,vem,comp)
618  integer :: ncartype
619  integer :: icar, i
620  real:: velocity,vem(:,:),comp(:,:)
621 ! ncartype Type of vehicle icar Type of vehicle
622 ! 11 Automoviles 1 Vehiculos ligeros a Gas
623 ! 12 Ligeros 2 Camionetas ligeras a Gaso
624 ! 13 Microbuses 2 Camionetas ligeras a Gaso
625 ! 13 Microbuses 5 Camiones Pesados a Gasoli
626 ! 14 Ruta 100 3 Camiones ligeros a diesel
627 ! 15 Otros camiones 3 Camiones ligeros a diesel
628 ! 16 Medianos 3 Camiones ligeros a diesel
629 ! 17 Pesados 4 Vehiculos pesados a diesel
630 ! 18 Camiones mpales. 3 Camiones ligeros a diesel
631 !
632  icar = 0
633  if( ncartype .eq. 11) icar =1
634  if( ncartype .eq. 12) icar =2
635  if( ncartype .eq. 13) icar =5
636  if( ncartype .eq. 14) icar =3
637  if( ncartype .eq. 15) icar =3
638  if( ncartype .eq. 16) icar =3
639  if( ncartype .eq. 17) icar =4
640  if( ncartype .eq. 18) icar =3
641  if (icar .eq. 0) then
642  write(6,*)'Invalid car type',ncartype
643  stop
644  end if
645  !..
646  i=0
647  if(velocity.le.vem(icar,2)) i=1
648  if(velocity.gt.vem(icar,2).and.velocity.le.vem(icar,3)) i=2
649  if(velocity.gt.vem(icar,3).and.velocity.le.vem(icar,4)) i=3
650  if(velocity.gt.vem(icar,4).and.velocity.le.vem(icar,5)) i=4
651  if(velocity.gt.vem(icar,5).and.velocity.le.vem(icar,6)) i=5
652  if(velocity.gt.vem(icar,6)) i=5
653  if (i.eq. 0) then
654  write(6,200)velocity
655  stop
656  end if
657 !..
658  emisfac2= comp(icar,i) +(velocity-vem(icar,i)) * &
659  (comp(icar,i+1)-comp(icar,i))/(vem(icar,i+1)-vem(icar,i))
660 !..
661 200 format('Invalid velocity',e10.1)
662  return
