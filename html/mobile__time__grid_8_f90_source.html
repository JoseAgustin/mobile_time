106  subroutine utm_2_ll(utmx,utmy,utmz,lat,long)
107  implicit none
108  integer,intent(IN):: utmZ
109  real,intent (IN) :: utmx,utmy
110  real,intent (OUT) ::lat,long
111  real:: utmym,dlong,dlongp,dlongx
112  real:: DEGRAD,latx
113 !
114 ! THIS SUBROUTINE CONVERTS from UTM coordinates
115 !
116  DATA degrad/0.017453/
117 
118  latx = utmy / 111.
119  dlongx = (utmx - 500.) / ((111.226 + 0.0053 * latx) * &
120  (cos(degrad * latx)))
121  utmym = utmy - (3187. * sin(2. * degrad * latx) * &
122  (1. -cos(degrad * dlongx)))
123  lat = (utmym - 2.41 - 0.00903 * latx * latx) / 110.270
124  dlongp = (utmx - 500.) / ((111.226 + 0.0053 * lat) * &
125  (cos(degrad * lat)))
126  long = -(180 - (6 * utmz) + 3 - dlongp )
127 
128  end subroutine utm_2_ll

302 subroutine calcula_emision
303  implicit none
304  integer:: ntime,isp,n,m
305  integer:: indx
306  real :: sl,fcorr,ffr
307  real :: temp
308  write(6,180)
309  do n = ntd ,1,-1 !Main LOOP initialization
310  do m = 1,nint
311  if(id_source_td(n).eq.id_source_int(m).and.geo_type_int(m).lt.2&
312  .and. geometry_type(m) .lt.2 ) then
313  indx = id_grid_int(m)
314  do ntime=1,nhr
315  ! for EPA
316  emiss_factor(1)= emisfac2(veh_type(n),cars_speed(ntime,n),ef_speed ,ef_hc )
317  emiss_factor(2)= emisfac2(veh_type(n),cars_speed(ntime,n),ef_speed ,ef_co )
318  emiss_factor(3)= emisfac2(veh_type(n),cars_speed(ntime,n),ef_speed ,ef_no )
319  emiss_factor(5)= emisfac2(veh_type(n),cars_speed(ntime,n),ef_speed ,ef_so2)
320  ! Emissions factor for cold start
321  emis_fact_cold(1)= emisfac2(veh_type(n),cars_speed(ntime,n),ef_speed_cold,ef_hc_cold)
322  emis_fact_cold(2)= emisfac2(veh_type(n),cars_speed(ntime,n),ef_speed_cold,ef_co_cold)
323  emis_fact_cold(3)= emisfac2(veh_type(n),cars_speed(ntime,n),ef_speed_cold,ef_no_cold)
324  emis_fact_cold(5)= emiss_factor(5)
325  if (veh_type(n).ge. 14 ) then
326  emiss_factor(4) =emiss_factor(1)
327  emis_fact_cold(4) =emis_fact_cold(1)
328  emiss_factor(1) = 0.0
329  emis_fact_cold(1) = 0.0
330  else
331  emiss_factor(4) = 0.0
332  emis_fact_cold(4) = 0.0
333  end if
334 !..
335 ! ---------- Localization of the viality lenght ----------
336 !..
337  call viality(geometry_type(m),id_source_int(m),geometry_type2,id_source_att,&
338  source_type,source_size,fcor,f_cold_engine_car,ntime,fcorr,ffr,sl)
339  sl = sl*r_weight(m)
340 ! ---------- Computation of the emissions for each specie
341  do isp = 1,nspc ! 1 HC(gasoline), 2 CO, 3 NOx, 4 HC(Diesel),5 SO2
342 !..
343  temp =(number_cars(ntime,n)*sl*emiss_factor(isp)*(1.0-ffr)+ &
344  number_cars(ntime,n)*sl*emis_fact_cold(isp)*ffr)*fcorr
345 !.. Xing is 7.5% of the emission of segment viality
346  if (geo_type_int(m).eq.2) temp=temp*0.075
347 !..
348  emision(indx,ntime,isp,id_time_period(n))= temp/3600.0 &
349  + emision(indx,ntime,isp,id_time_period(n))
350 !..
351 ! ---------- Computation of dayly emissions EDAY ----------
352 !
353  eday(indx,isp,id_time_period(n))= temp/3600.0 &
354  + eday(indx,isp,id_time_period(n))
355  end do ! i nspc
356  end do ! m nint
357  end if
358  end do ! ntd
359  end do
360 180 format(9x,'++++++ STARTS EMISSIONS COMPUTATIONS',4x,'******')
361 
362 end subroutine calcula_emision

701  real function emisfac2(ncartype,velocity,vem,comp)
702  integer :: ncartype
703  integer :: icar, i
704  real:: velocity,vem(:,:),comp(:,:)
705 ! ncartype Type of vehicle icar Type of vehicle
706 ! 11 Automoviles 1 Vehiculos ligeros a Gas
707 ! 12 Ligeros 2 Camionetas ligeras a Gaso
708 ! 13 Microbuses 2 Camionetas ligeras a Gaso
709 ! 13 Microbuses 5 Camiones Pesados a Gasoli
710 ! 14 Ruta 100 3 Camiones ligeros a diesel
711 ! 15 Otros camiones 3 Camiones ligeros a diesel
712 ! 16 Medianos 3 Camiones ligeros a diesel
713 ! 17 Pesados 4 Vehiculos pesados a diesel
714 ! 18 Camiones mpales. 3 Camiones ligeros a diesel
715 !
716  icar = 0
717  if( ncartype .eq. 11) icar =1
718  if( ncartype .eq. 12) icar =2
719  if( ncartype .eq. 13) icar =5
720  if( ncartype .eq. 14) icar =3
721  if( ncartype .eq. 15) icar =3
722  if( ncartype .eq. 16) icar =3
723  if( ncartype .eq. 17) icar =4
724  if( ncartype .eq. 18) icar =3
725  if (icar .eq. 0) then
726  write(6,*)'Invalid car type',ncartype
727  stop
728  end if
729  !..
730  i=0
731  if(velocity.le.vem(icar,2)) i=1
732  if(velocity.gt.vem(icar,2).and.velocity.le.vem(icar,3)) i=2
733  if(velocity.gt.vem(icar,3).and.velocity.le.vem(icar,4)) i=3
734  if(velocity.gt.vem(icar,4).and.velocity.le.vem(icar,5)) i=4
735  if(velocity.gt.vem(icar,5).and.velocity.le.vem(icar,6)) i=5
736  if(velocity.gt.vem(icar,6)) i=5
737  if (i.eq. 0) then
738  write(6,200)velocity
739  stop
740  end if
741 !..
742  emisfac2= comp(icar,i) +(velocity-vem(icar,i)) * &
743  (comp(icar,i+1)-comp(icar,i))/(vem(icar,i+1)-vem(icar,i))
744 !..
745 200 format('Invalid speed',e10.1)
746  return
747 !*********************************************************************
748 !********* END OF FUNCTION EMISFAC2 *********
749 !*********************************************************************
750  end
